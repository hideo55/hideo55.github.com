シンタックスハイライト

mosho.plugin({
  name: "scaleActiveSlide",
  preShow: function (evt) {
    prv = evt.detail.prevSlide;
    nxt = evt.detail.nextSlide;
    if (prv) { prv.scale(0.25) }
    if (nxt) { nxt.scale(4) }
  }
});
